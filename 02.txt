CREATE TEMPORARY TABLE big_video (
	id INT,
	url VARCHAR(130),
	owner_id INT,
	size INT
);
INSERT INTO big_video
	SELECT
        video.id,
        video.url,
        video.owner_id,
        video.size
	FROM video
    ORDER BY video.size DESC
    LIMIT 10;

SELECT
	big_video.id,
	users.first_name,
	users.last_name,
	photo.url,
	big_video.url,
	big_video.size
FROM big_video
JOIN users
	ON big_video.owner_id = users.id
JOIN photo
	ON photo.id = users.main_photo_id;
